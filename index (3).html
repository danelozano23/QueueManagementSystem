⚙️ Google Sheets Setup

Connect your queue system to a live Google Sheet in 3 steps

Step 1 — Create the Google Apps Script

Open script.google.com, create a new project, paste the code below, then click Deploy → New Deployment → Web App. Set "Execute as" to Me and "Who has access" to Anyone. Copy the Web App URL.

// RHU Queue — Google Apps Script Backend // Paste this entire file into script.google.com const SHEET_NAME = 'Queue'; function doGet(e) { return handleRequest(e); } function doPost(e) { return handleRequest(e); } function handleRequest(e) { const params = e.parameter; const action = params.action; let result; try { if (action === 'ping') result = { ok: true, message: 'Connected!' }; else if (action === 'append') result = appendRow(params); else if (action === 'update') result = updateRow(params); else if (action === 'read') result = readAll(); else if (action === 'delete') result = deleteRow(params); else result = { ok: false, message: 'Unknown action' }; } catch(err) { result = { ok: false, message: err.toString() }; } return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } function getSheet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); sheet.appendRow([ 'ID','Queue #','Code','Service','First Name','Last Name', 'Date of Birth','Sex','Civil Status','Contact No','PhilHealth No', 'Barangay','Visit Type','Chief Complaint','Notes', 'Status','Registered Time','Date','Last Updated' ]); sheet.setFrozenRows(1); // Style header const header = sheet.getRange(1,1,1,19); header.setBackground('#074d37'); header.setFontColor('#ffffff'); header.setFontWeight('bold'); sheet.setColumnWidth(14, 300); sheet.setColumnWidth(15, 200); } return sheet; } function appendRow(p) { const sheet = getSheet(); const now = new Date().toLocaleString('en-PH'); sheet.appendRow([ p.id, p.queueNum, p.code, p.service, p.fname, p.lname, p.dob, p.sex, p.civil, p.contact, p.philhealth || '', p.barangay, p.visitType, p.complaint, p.notes || '', 'Waiting', p.regTime, p.regDate, now ]); return { ok: true, message: 'Row added' }; } function updateRow(p) { const sheet = getSheet(); const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (String(data[i][0]) === String(p.id)) { const now = new Date().toLocaleString('en-PH'); // Status is column 16 (index 15), Last Updated column 19 (index 18) sheet.getRange(i+1, 16).setValue( p.status.charAt(0).toUpperCase() + p.status.slice(1).replace('-',' ') ); sheet.getRange(i+1, 19).setValue(now); return { ok: true, message: 'Row updated' }; } } return { ok: false, message: 'Row not found' }; } function readAll() { const sheet = getSheet(); const data = sheet.getDataRange().getValues(); if (data.length <= 1) return { ok: true, rows: [] }; const headers = data[0]; const rows = data.slice(1).map(row => { const obj = {}; headers.forEach((h, i) => obj[h] = row[i]); return obj; }); return { ok: true, rows }; } function deleteRow(p) { const sheet = getSheet(); const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (String(data[i][0]) === String(p.id)) { sheet.deleteRow(i + 1); return { ok: true, message: 'Row deleted' }; } } return { ok: false, message: 'Row not found' }; }

Step 2 — Paste Your Web App URL

  1. In Apps Script, click Deploy → New Deployment
  2. Choose type: Web App
  3. Set Execute as: Me and Who has access: Anyone
  4. Click Deploy and authorize permissions when prompted
  5. Copy the Web App URL (starts with https://script.google.com/macros/s/...)

🎉 All Set!

Your RHU Queue System is now connected to Google Sheets. Every registration and status change will sync automatically. Your data is accessible anytime at script.google.com → Your Project → Spreadsheet.

  1. Data syncs to Google Sheets on every registration, call, and status update
  2. Open the Queue Board to see all patients and their live statuses
  3. Use Export to Excel on the Queue Board to download a local copy anytime
  4. To reconfigure, open the burger menu and tap ⚙ Settings
